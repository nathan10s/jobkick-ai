    copyBtn.addEventListener('click', async () => {
      const text = output.textContent.trim();
      if (!text || text.startsWith('Fill the form')) {
        output.textContent = 'Generate some CV text first, then you can copy it.';
        return;
      }
      try {
        await navigator.clipboard.writeText(text);
        const original = output.textContent;
        output.textContent = original + '\n\n[Copied to clipboard ‚úÖ]';
      } catch (err) {
        const original = output.textContent;
        output.textContent = original + '\n\n[Could not copy automatically ‚Äì please select and copy manually.]';
      }
    });
